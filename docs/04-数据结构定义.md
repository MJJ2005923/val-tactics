# 04 — 数据结构定义

## TypeScript 类型

```typescript
// 技能类型
type AbilityType = 'smoke' | 'flash' | 'damage' | 'recon' | 'control' | 'heal' | 'mobility'

// 技能
interface Ability {
  id: string          // 唯一标识，如 "sage-barrier-orb"
  name: string        // 中文名称
  nameEn: string      // 英文名称
  key: 'C' | 'Q' | 'E' | 'X'  // 默认按键（大招为 X）
  type: AbilityType
  icon: string        // 图标路径
  description: string // 详细介绍
  usage: string       // 释放方式描述
}

// 英雄
interface Agent {
  id: string          // 唯一标识，如 "sage"
  name: string        // 中文名称
  nameEn: string      // 英文名称
  icon: string        // 头像路径
  role: string        // 定位：控场者/决斗者/先锋/哨卫
  abilities: Ability[]
}

// 地图
interface MapInfo {
  id: string          // 唯一标识
  name: string        // 中文名称
  nameEn: string      // 英文名称
  image: string       // 地图图片路径
}

// 地图上的技能标记
interface Marker {
  id: string          // 标记唯一 ID
  abilityId: string   // 关联的技能 ID
  agentId: string     // 关联的英雄 ID
  x: number           // 地图上的相对 X 坐标 (0-1)
  y: number           // 地图上的相对 Y 坐标 (0-1)
  step: number        // 步骤编号（从 1 开始）
  time: number        // 时间点（秒）
  note: string        // 备注文字
}

// 战术模板
interface Template {
  id: string          // 模板唯一 ID
  name: string        // 模板名称
  mapId: string       // 关联地图
  createdAt: number   // 创建时间戳
  updatedAt: number   // 更新时间戳
  markers: Marker[]
}
```

## IndexedDB 表结构

数据库名：`val-tactics-db`，版本：1

### 表：templates

| 字段 | 类型 | 说明 |
|------|------|------|
| id | string (主键) | UUID |
| name | string | 模板名称 |
| mapId | string | 地图 ID |
| createdAt | number | 创建时间戳 |
| updatedAt | number | 更新时间戳 |
| markers | Marker[] | 技能标记数组 |
